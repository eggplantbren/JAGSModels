model
{
	mu ~ dnorm(0.75, 1.0)
	log_timescale ~ dunif(-10.0, 10.0)
	timescale <- exp(log_timescale)
	alpha <- exp(-1.0/timescale)

	log_sig ~ dunif(-10.0, 10.0)
	sig <- exp(log_sig)
	tau <- pow(sig, -2)

	for(i in 2:N)
	{
		mean[i] <- mu + alpha*(y[i-1] - mu)
		y[i] ~ dnorm(mean[i], tau)
	}

	y_future[1] <- y[N]
	for(i in 2:1000)
	{
		mean_future[i] <- mu + alpha*(y_future[i-1] - mu)
		y_future[i] ~ dnorm(mean_future[i], tau)
	}

}
